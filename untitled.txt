  const charsDiv = document.getElementById('characters');
  const textInput = document.getElementById('textInput');
  const speakBtn = document.getElementById('speakBtn');

  let selectedChar = characters[0];

  function renderCharacters() {
    charsDiv.innerHTML = '';
    characters.forEach(c => {
      const card = document.createElement('div');
      card.className = 'char-card' + (c.id === selectedChar.id ? ' selected' : '');
      card.style.border = `2px solid ${c.color}`;
      card.innerHTML = `<div class="char-avatar">${c.avatar}</div><div>${c.name}</div>`;
      card.onclick = () => {
        selectedChar = c;
        renderCharacters();
      };
      charsDiv.appendChild(card);
    });
  }

  speakBtn.onclick = () => {
    const text = textInput.value.trim();
    if (!text) {
      alert('Please type something!');
      return;
    }
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'en-US';
    utterance.voice = speechSynthesis.getVoices().find(v => v.lang === 'en-US') || null;
    speechSynthesis.speak(utterance);
  };

  renderCharacters();
</script>
